#### 迭代器必须有： 1.可迭代的（iterator）属性，2.next()函数并且执行后返回{done:Boolen 值,value:迭代值}

        let something = (function () {
            let nextVal;
            return {
                [Symbol.iterator]: function () { return this; },
                next: function () {
                    if (nextVal === undefined) {
                        nextVal = 1;
                    } else {
                        nextVal = (3 * nextVal + 6)
                    }
                    return { done: false, value: nextVal }
                }
            }
        })()
        for (let v of something) {
            console.log(v)
            if (v > 500) {
                break
            }
        }
